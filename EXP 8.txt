-------------------------Without Input Taking From The User-----------------------------

CASE 1


SET SERVEROUTPUT ON;

DECLARE
    s1 NUMBER;
    s2 NUMBER;
    s3 NUMBER;
    s4 NUMBER;
    total NUMBER;
    per NUMBER;
BEGIN
    s1 := 75;
    s2 := 45;
    s3 := 52;
    s4 := 75;
    IF s1 < 50 THEN
        DBMS_OUTPUT.PUT_LINE('Result: FAILED IN s1');
    ELSIF s2 < 50 THEN
        DBMS_OUTPUT.PUT_LINE('Result: FAILED IN s2');
    ELSIF s3 < 50 THEN
        DBMS_OUTPUT.PUT_LINE('Result: FAILED IN s3');
    ELSIF s4 < 50 THEN
        DBMS_OUTPUT.PUT_LINE('Result: FAILED IN s4');
    ELSE
        total := s1 + s2 + s3 + s4;
        per := (total/400)*100;

        DBMS_OUTPUT.PUT_LINE('Result: PASS');
        DBMS_OUTPUT.PUT_LINE('Total Marks: ' || total);
        DBMS_OUTPUT.PUT_LINE('Percentage: ' || per);

        IF per > 75 THEN
            DBMS_OUTPUT.PUT_LINE('Division: DISTINCTION');
        ELSIF per > 60 THEN
            DBMS_OUTPUT.PUT_LINE('Division: FIRST DIVISION');
        ELSIF per > 50 THEN
            DBMS_OUTPUT.PUT_LINE('Division: SECOND DIVISION');
        ELSE
            DBMS_OUTPUT.PUT_LINE('Division: PASS');
        END IF;
    END IF;
END;
/


case 1 output
Result: FAILED IN s2

PL/SQL procedure successfully completed.
